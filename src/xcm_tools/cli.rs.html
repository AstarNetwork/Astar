19/// Astar XCM tools.
20#[derive(Debug, clap::Parser)]
21#[clap(subcommand_required = true)]
22pub struct Cli {
23    /// Possible subcommand with parameters.
24    #[clap(subcommand)]
25    pub subcommand: Option<Subcommand>,
26}
27
28/// Possible subcommands of the main binary.
29#[derive(Debug, clap::Subcommand)]
30pub enum Subcommand {
31    /// Prints relay-chain SS58 account Id
32    RelayChainAccount,
33    /// Prints parachains sovereign SS58 account Id.
34    SovereignAccount(SovereignAccountCmd),
35    /// Prints AssetId for desired parachain asset.
36    AssetId(AssetIdCmd),
37    /// Prints derived remote SS58 account for the derived multilocation.
38    RemoteAccount(RemoteAccountCmd),
39}
40
41/// Helper that prints AccountId of parachain.
42#[derive(Debug, clap::Parser)]
43pub struct SovereignAccountCmd {
44    /// Print address for sibling parachain [child by default].
45    #[clap(short)]
46    pub sibling: bool,
47
48    /// Target ParaId.
49    pub parachain_id: u32,
50}
51
52/// Helper that prints AssetId for sibling parachain asset.
53#[derive(Debug, clap::Parser)]
54pub struct AssetIdCmd {
55    /// External AssetId [relay by default].
56    #[clap(default_value = "340282366920938463463374607431768211455")]
57    pub asset_id: u128,
58}
59
60/// Helper that prints the derived AccountId32 value for the multilocation.
61#[derive(Debug, clap::Parser)]
62pub struct RemoteAccountCmd {
63    /// Parachain id in case sender is from a sibling parachain.
64    #[clap(short, long, default_value = None)]
65    pub parachain_id: Option<u32>,
66    /// Public key (SS58 or H160) in hex format. Must be either 32 or 20 bytes long.
67    #[clap(short, long)]
68    pub account_key: AccountWrapper,
69}
70
71#[derive(Debug, Clone, Copy)]
72pub enum AccountWrapper {
73    SS58([u8; 32]),
74    H160([u8; 20]),
75}
76
77impl std::str::FromStr for AccountWrapper {
78    type Err = String;
79
80    fn from_str(account_pub_key: &str) -> Result<Self, Self::Err> {
81        if let Some(rest) = account_pub_key.strip_prefix("0x") {
82            if let Some(pos) = rest.chars().position(|c| !c.is_ascii_hexdigit()) {
83                Err(format!(
84					"Expected account public key in hex format, found illegal hex character at position: {}",
85					2 + pos,
86				))
87            } else {
88                let account = hex::decode(rest).expect("Ensured in previous check it's hex; QED");
89                if rest.len() == 40 {
90                    Ok(AccountWrapper::H160(
91                        account
92                            .try_into()
93                            .expect("Ensured length in previous check; QED"),
94                    ))
95                } else if rest.len() == 64 {
96                    Ok(AccountWrapper::SS58(
97                        account
98                            .try_into()
99                            .expect("Ensured length in previous check; QED"),
100                    ))
101                } else {
102                    Err("Account key should be 20 or 32 bytes long".into())
103                }
104            }
105        } else {
106            Err("Account key should start with '0x'".into())
107        }
108    }
109}
