19use crate::*;
20use astar_primitives::{evm::EVM_REVERT_CODE, genesis::GenesisAccount, parachain::SHIBUYA_ID};

22/// Provides the JSON representation of predefined genesis config for given `id`.
23pub fn get_preset(id: &sp_genesis_builder::PresetId) -> Option<Vec<u8>> {
24    let genesis = match id.as_str() {
25        "development" => default_config(SHIBUYA_ID),
26        _ => return None,
27    };
28    Some(
29        serde_json::to_string(&genesis)
30            .expect("serialization to json is expected to work. qed.")
31            .into_bytes(),
32    )
33}

35/// Get the default genesis config for the Shibuya runtime.
36pub fn default_config(para_id: u32) -> serde_json::Value {
37    let alice = GenesisAccount::<sr25519::Public>::from_seed("Alice");
38    let bob = GenesisAccount::<sr25519::Public>::from_seed("Bob");
39    let charlie = GenesisAccount::<sr25519::Public>::from_seed("Charlie");
40    let dave = GenesisAccount::<sr25519::Public>::from_seed("Dave");
41    let eve = GenesisAccount::<sr25519::Public>::from_seed("Eve");
42
43    let authorities = vec![&alice, &bob];
44    let accounts = vec![&alice, &bob, &charlie, &dave, &eve]
45        .iter()
46        .map(|x| x.account_id())
47        .collect::<Vec<_>>();
48
49    let balances = accounts
50        .iter()
51        .chain(
52            vec![
53                TreasuryPalletId::get().into_account_truncating(),
54                CommunityTreasuryPalletId::get().into_account_truncating(),
55            ]
56            .iter(),
57        )
58        .map(|x| (x.clone(), 1_000_000_000 * SBY))
59        .collect::<Vec<_>>();
60
61    let config = RuntimeGenesisConfig {
62        system: Default::default(),
63        sudo: SudoConfig {
64            key: Some(alice.account_id()),
65        },
66        parachain_info: ParachainInfoConfig {
67            parachain_id: para_id.into(),
68            ..Default::default()
69        },
70        balances: BalancesConfig {
71            balances,
72            ..Default::default()
73        },
74        vesting: VestingConfig { vesting: vec![] },
75        session: SessionConfig {
76            keys: authorities
77                .iter()
78                .map(|x| {
79                    (
80                        x.account_id(),
81                        x.account_id(),
82                        SessionKeys {
83                            aura: x.pub_key().into(),
84                        },
85                    )
86                })
87                .collect::<Vec<_>>(),
88            ..Default::default()
89        },
90        aura: AuraConfig {
91            authorities: vec![],
92        },
93        aura_ext: Default::default(),
94        collator_selection: CollatorSelectionConfig {
95            desired_candidates: 32,
96            candidacy_bond: 32_000 * SBY,
97            invulnerables: authorities
98                .iter()
99                .map(|x| x.account_id())
100                .collect::<Vec<_>>(),
101        },
102        evm: EVMConfig {
103            // We need _some_ code inserted at the precompile address so that
104            // the evm will actually call the address.
105            accounts: Precompiles::used_addresses_h160()
106                .map(|addr| {
107                    (
108                        addr,
109                        fp_evm::GenesisAccount {
110                            nonce: Default::default(),
111                            balance: Default::default(),
112                            storage: Default::default(),
113                            code: EVM_REVERT_CODE.into(),
114                        },
115                    )
116                })
117                .collect(),
118            ..Default::default()
119        },
120        evm_chain_id: EVMChainIdConfig {
121            chain_id: 0x51,
122            ..Default::default()
123        },
124        ethereum: Default::default(),
125        polkadot_xcm: Default::default(),
126        assets: Default::default(),
127        parachain_system: Default::default(),
128        transaction_payment: Default::default(),
129        dapp_staking: DappStakingConfig {
130            reward_portion: vec![
131                Permill::from_percent(40),
132                Permill::from_percent(30),
133                Permill::from_percent(20),
134                Permill::from_percent(10),
135            ],
136            slot_distribution: vec![
137                Permill::from_percent(10),
138                Permill::from_percent(20),
139                Permill::from_percent(30),
140                Permill::from_percent(40),
141            ],
142            // percentages below are calculated based on a total issuance at the time when dApp staking v3 was launched (147M)
143            tier_thresholds: vec![
144                TierThreshold::DynamicPercentage {
145                    percentage: Perbill::from_parts(20_000), // 0.0020%
146                    minimum_required_percentage: Perbill::from_parts(17_000), // 0.0017%
147                    maximum_possible_percentage: Perbill::from_percent(100),
148                },
149                TierThreshold::DynamicPercentage {
150                    percentage: Perbill::from_parts(13_000), // 0.0013%
151                    minimum_required_percentage: Perbill::from_parts(10_000), // 0.0010%
152                    maximum_possible_percentage: Perbill::from_percent(100),
153                },
154                TierThreshold::DynamicPercentage {
155                    percentage: Perbill::from_parts(5_400), // 0.00054%
156                    minimum_required_percentage: Perbill::from_parts(3_400), // 0.00034%
157                    maximum_possible_percentage: Perbill::from_percent(100),
158                },
159                TierThreshold::FixedPercentage {
160                    required_percentage: Perbill::from_parts(1_400), // 0.00014%
161                },
162            ],
163            slots_per_tier: vec![10, 20, 30, 40],
164            safeguard: Some(false),
165            ..Default::default()
166        },
167        inflation: Default::default(),
168        oracle_membership: OracleMembershipConfig {
169            members: vec![alice.account_id(), bob.account_id()]
170                .try_into()
171                .expect("Assumption is that at least two members will be allowed."),
172            ..Default::default()
173        },
174        price_aggregator: PriceAggregatorConfig {
175            circular_buffer: vec![CurrencyAmount::from_rational(5, 10)]
176                .try_into()
177                .expect("Must work since buffer should have at least a single value."),
178        },
179        council_membership: CouncilMembershipConfig {
180            members: accounts
181                .clone()
182                .try_into()
183                .expect("Should support at least 5 members."),
184            phantom: Default::default(),
185        },
186        technical_committee_membership: TechnicalCommitteeMembershipConfig {
187            members: accounts[..3]
188                .to_vec()
189                .try_into()
190                .expect("Should support at least 3 members."),
191            phantom: Default::default(),
192        },
193        community_council_membership: CommunityCouncilMembershipConfig {
194            members: accounts
195                .try_into()
196                .expect("Should support at least 5 members."),
197            phantom: Default::default(),
198        },
199        council: Default::default(),
200        technical_committee: Default::default(),
201        community_council: Default::default(),
202        democracy: Default::default(),
203        treasury: Default::default(),
204        community_treasury: Default::default(),
205        safe_mode: Default::default(),
206        tx_pause: Default::default(),
207    };
208
209    serde_json::to_value(&config).expect("Could not build genesis config.")
210}
