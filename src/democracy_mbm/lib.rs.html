69/// Progressive migration state to keep track of progress
70#[derive(Clone, Eq, PartialEq, Encode, Decode, MaxEncodedLen)]
71pub enum MigrationStep<T: pallet_democracy::Config> {
72    /// Migrating referendum info
73    ReferendumInfo(ReferendumIndex),
74    /// Finished Migrating referendum info, starting migration VotingOf
75    StartingVotingOf,
76    /// Migrating VotingOf
77    VotingOf(<T as frame_system::Config>::AccountId),
78    /// Finished all migrations
79    Finished,
80}
81
82/// Stores the migration step and the block number at which the migration started
83#[derive(Clone, Eq, PartialEq, Encode, Decode, MaxEncodedLen)]
84pub struct MigrationState<T: pallet_democracy::Config> {
85    pub step: MigrationStep<T>,
86    pub start_block: u32,
87}
88
89type StepResultOf<T> = (MigrationStep<T>, bool);

91pub struct DemocracyMigrationV1ToV2<T, W: weights::WeightInfo>(PhantomData<(T, W)>);

93impl<
94        T: pallet_democracy::Config + frame_system::Config + pallet::Config,
95        W: weights::WeightInfo,
96    > SteppedMigration for DemocracyMigrationV1ToV2<T, W>
97{
98    type Cursor = MigrationState<T>;
99    // Without the explicit length here the construction of the ID would not be infallible.
100    type Identifier = MigrationId<20>;
101
102    /// The identifier of this migration. Which should be globally unique.
103    fn id() -> Self::Identifier {
104        MigrationId {
105            pallet_id: *PALLET_MIGRATIONS_ID,
106            version_from: 1,
107            version_to: 2,
108        }
109    }
110
111    fn step(
112        mut cursor: Option<Self::Cursor>,
113        meter: &mut WeightMeter,
114    ) -> Result<Option<Self::Cursor>, SteppedMigrationError> {
115        // Check that we have enough weight for at least the next step. If we don't, then the
116        // migration cannot be complete.
117        let required = match &cursor {
118            Some(state) => Self::required_weight(&state.step),
119            // Worst case weight for `migration_voting_of`.
120            None => W::migration_voting_of(),
121        };
122        if meter.remaining().any_lt(required) {
123            return Err(SteppedMigrationError::InsufficientWeight { required });
124        }
125
126        let mut referendum_entries = 0u32;
127        let mut referendum_migrated = 0u32;
128        let mut voting_entries = 0u32;
129        let mut voting_migrated = 0u32;
130
131        // We loop here to do as much progress as possible per step.
132        loop {
133            // Check that we would have enough weight to perform this step in the worst case
134            // scenario.
135            let required_weight = match &cursor {
136                Some(state) => Self::required_weight(&state.step),
137                // Worst case weight for `migration_voting_of`.
138                None => W::migration_voting_of(),
139            };
140            if !meter.can_consume(required_weight) {
141                break;
142            }
143
144            let next = match &cursor {
145                // At first, migrate referendums and get the current block number to set at start_block
146                None => {
147                    let block_number = MigrationBlockNumber::<T>::get();
148                    Self::process_migration_result(
149                        Self::migrate_referendum_info(None, block_number),
150                        &mut referendum_migrated,
151                        &mut referendum_entries,
152                        block_number,
153                    )
154                }
155                // Migrate any remaining referendums
156                Some(MigrationState {
157                    step: MigrationStep::ReferendumInfo(maybe_last_referendum),
158                    start_block,
159                }) => Self::process_migration_result(
160                    Self::migrate_referendum_info(Some(maybe_last_referendum), *start_block),
161                    &mut referendum_migrated,
162                    &mut referendum_entries,
163                    *start_block,
164                ),
165                // After the last referendum was migrated, start migrating VotingOf
166                Some(MigrationState {
167                    step: MigrationStep::StartingVotingOf,
168                    start_block,
169                }) => Self::process_migration_result(
170                    Self::migrate_voting_of(None, *start_block),
171                    &mut voting_migrated,
172                    &mut voting_entries,
173                    *start_block,
174                ),
175                // Keep migrating VotingOf
176                Some(MigrationState {
177                    step: MigrationStep::VotingOf(maybe_last_vote),
178                    start_block,
179                }) => Self::process_migration_result(
180                    Self::migrate_voting_of(Some(maybe_last_vote), *start_block),
181                    &mut voting_migrated,
182                    &mut voting_entries,
183                    *start_block,
184                ),
185                Some(MigrationState {
186                    step: MigrationStep::Finished,
187                    ..
188                }) => {
189                    StorageVersion::new(Self::id().version_to as u16)
190                        .put::<pallet_democracy::Pallet<T>>();
191                    log::info!(target: LOG_TARGET, "Democracy MBM migration finished");
192                    return Ok(None);
193                }
194            };
195
196            cursor = Some(next);
197            meter.consume(required_weight);
198        }
199
200        log::info!(target: LOG_TARGET, "Iterated over {referendum_entries} referendum entries, migrated {referendum_migrated} referendums. Iterated over {voting_entries} votes entries, migrated {voting_migrated} votes");
201        Ok(cursor)
202    }
203
204    #[cfg(feature = "try-runtime")]
205    fn pre_upgrade() -> Result<Vec<u8>, sp_runtime::TryRuntimeError> {
206        try_runtime::pre_upgrade_body::<T>()
207    }
208
209    #[cfg(feature = "try-runtime")]
210    fn post_upgrade(state: Vec<u8>) -> Result<(), sp_runtime::TryRuntimeError> {
211        try_runtime::post_upgrade_body::<T>(state)
212    }
213}

215impl<
216        T: pallet_democracy::Config + frame_system::Config + pallet::Config,
217        W: weights::WeightInfo,
218    > DemocracyMigrationV1ToV2<T, W>
219{
220    fn required_weight(state: &MigrationStep<T>) -> Weight {
221        match state {
222            MigrationStep::ReferendumInfo(_) => W::migration_referendum_info(),
223            MigrationStep::StartingVotingOf | MigrationStep::VotingOf(_) => {
224                W::migration_voting_of()
225            }
226            MigrationStep::Finished => Weight::zero(),
227        }
228    }
229
230    fn migrate_referendum_info(
231        maybe_last_key: Option<&ReferendumIndex>,
232        block_number: u32,
233    ) -> StepResultOf<T> {
234        let mut iter = if let Some(last_key) = maybe_last_key {
235            ReferendumInfoOf::<T>::iter_from_key(last_key)
236        } else {
237            ReferendumInfoOf::<T>::iter()
238        };
239
240        if let Some((last_key, mut ref_info)) = iter.next() {
241            match ref_info {
242                ReferendumInfo::Ongoing(ref mut status) => {
243                    // Double the blocks of the delay period
244                    status.delay = status
245                        .delay
246                        .saturated_into::<u32>()
247                        .saturating_mul(2)
248                        .into();
249
250                    // For the end time:
251                    // 1. Calculate remaining blocks until the original end
252                    let remaining_blocks = status
253                        .end
254                        .saturated_into::<u32>()
255                        .saturating_sub(block_number);
256
257                    // 2. Double the remaining blocks
258                    let doubled_remaining = remaining_blocks.saturating_mul(2);
259
260                    // 3. Add it to the current block number to get the new end
261                    status.end = block_number.saturating_add(doubled_remaining).into();
262
263                    ReferendumInfoOf::<T>::insert(&last_key, ref_info);
264
265                    // entry has been migrated
266                    return (MigrationStep::ReferendumInfo(last_key), true);
267                }
268                ReferendumInfo::Finished { .. } => {
269                    // continue;
270                }
271            }
272
273            (MigrationStep::ReferendumInfo(last_key), false)
274        } else {
275            (MigrationStep::StartingVotingOf, false)
276        }
277    }
278
279    fn migrate_voting_of(
280        maybe_last_key: Option<&T::AccountId>,
281        block_number: u32,
282    ) -> StepResultOf<T> {
283        let mut iter = if let Some(last_key) = maybe_last_key {
284            VotingOf::<T>::iter_from_key(last_key)
285        } else {
286            VotingOf::<T>::iter()
287        };
288
289        if let Some((last_key, mut voting)) = iter.next() {
290            match &mut voting {
291                Voting::Direct { prior, .. } | Voting::Delegating { prior, .. } => {
292                    let lock_amount = prior.locked();
293
294                    if !lock_amount.is_zero() {
295                        // 1. Calculate the remaining blocks
296                        // as the field block_number is private in PriorLock enum
297                        // it encodes the enum and decodes the 4 bytes (as it's an u32)
298                        let encoded = prior.encode();
299                        let unlock_block_number =
300                            u32::from_le_bytes([encoded[0], encoded[1], encoded[2], encoded[3]]);
301
302                        // Some past lock still remains if they haven't been unlocked
303                        // only migrate when they are in the future
304                        if unlock_block_number > block_number {
305                            let remaining_blocks = unlock_block_number
306                                .saturating_sub(block_number)
307                                .saturating_mul(2);
308                            let extended_time =
309                                block_number.saturating_add(remaining_blocks).into();
310
311                            // 2. Clean the lock by setting block number and balance to 0
312                            prior.rejig(u32::MAX.into());
313
314                            // 3. Save the lock with migrated values
315                            prior.accumulate(extended_time, lock_amount);
316
317                            VotingOf::<T>::insert(&last_key, voting);
318
319                            // entry has been migrated
320                            return (MigrationStep::VotingOf(last_key), true);
321                        }
322                    }
323                }
324            }
325
326            (MigrationStep::VotingOf(last_key), false)
327        } else {
328            (MigrationStep::Finished, false)
329        }
330    }
331
332    fn process_migration_result(
333        result: StepResultOf<T>,
334        migrate_counter: &mut u32,
335        entries_counter: &mut u32,
336        start_block: u32,
337    ) -> MigrationState<T> {
338        let (step, was_updated) = result;
339        if was_updated {
340            migrate_counter.saturating_inc();
341        }
342        entries_counter.saturating_inc();
343        MigrationState { step, start_block }
344    }
345}

347pub struct DemocracyMigrationSaveMigrationBlock<T>(PhantomData<T>);
348
349impl<T: Config> OnRuntimeUpgrade for DemocracyMigrationSaveMigrationBlock<T> {
350    fn on_runtime_upgrade() -> Weight {
351        let block_number = frame_system::Pallet::<T>::block_number().saturated_into::<u32>();
352        MigrationBlockNumber::<T>::set(block_number);
353        T::DbWeight::get().reads_writes(0, 1)
354    }
355}
